PRAGMA foreign_keys = ON;

-- =========================
-- Core tables
-- =========================

CREATE TABLE USER (
  userID INTEGER PRIMARY KEY,
  username TEXT NOT NULL UNIQUE,
  email TEXT NOT NULL UNIQUE,
  passwordHash TEXT NOT NULL,
  subscriptionTier TEXT NOT NULL
    CHECK (subscriptionTier IN ('Free','Premium','Family')),
  dateJoined TEXT NOT NULL  -- store as 'YYYY-MM-DD'
);

CREATE TABLE ARTIST (
  artistID INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  country TEXT,
  primaryGenre TEXT,
  bio TEXT
);

CREATE TABLE ALBUM (
  albumID INTEGER PRIMARY KEY,
  title TEXT NOT NULL,
  releaseDate TEXT,  -- 'YYYY-MM-DD'
  albumType TEXT NOT NULL
    CHECK (albumType IN ('Single','EP','LP')),
  coverImageURL TEXT,
  artistID INTEGER NOT NULL,
  FOREIGN KEY (artistID) REFERENCES ARTIST(artistID)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);

CREATE TABLE SONG (
  songID INTEGER PRIMARY KEY,
  title TEXT NOT NULL,
  duration INTEGER NOT NULL CHECK (duration >= 0), -- seconds
  trackNumber INTEGER NOT NULL CHECK (trackNumber >= 1),
  isExplicit INTEGER NOT NULL CHECK (isExplicit IN (0,1)),
  albumID INTEGER NOT NULL,
  FOREIGN KEY (albumID) REFERENCES ALBUM(albumID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

CREATE TABLE PLAYLIST (
  playlistID INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  visibility TEXT NOT NULL CHECK (visibility IN ('Public','Private')),
  createdDate TEXT NOT NULL, -- 'YYYY-MM-DD'
  userID INTEGER NOT NULL,
  FOREIGN KEY (userID) REFERENCES USER(userID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

CREATE TABLE LISTEN_HISTORY (
  historyID INTEGER PRIMARY KEY,
  timestamp TEXT NOT NULL, -- 'YYYY-MM-DD HH:MM:SS'
  playDuration INTEGER NOT NULL CHECK (playDuration >= 0),
  userID INTEGER NOT NULL,
  songID INTEGER NOT NULL,
  FOREIGN KEY (userID) REFERENCES USER(userID)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  FOREIGN KEY (songID) REFERENCES SONG(songID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

-- =========================
-- Association tables (no M:N in ERD)
-- =========================

CREATE TABLE USER_ARTIST (
  userID INTEGER NOT NULL,
  artistID INTEGER NOT NULL,
  followedDate TEXT NOT NULL, -- 'YYYY-MM-DD'
  PRIMARY KEY (userID, artistID),
  FOREIGN KEY (userID) REFERENCES USER(userID)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  FOREIGN KEY (artistID) REFERENCES ARTIST(artistID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

CREATE TABLE SONG_ARTIST (
  songID INTEGER NOT NULL,
  artistID INTEGER NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('Primary','Featured','Producer')),
  PRIMARY KEY (songID, artistID),
  FOREIGN KEY (songID) REFERENCES SONG(songID)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  FOREIGN KEY (artistID) REFERENCES ARTIST(artistID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

-- If you want to allow the same song multiple times in a playlist,
-- change PK to (playlistID, position) instead.
CREATE TABLE PLAYLIST_SONG (
  UNIQUE (playlistID, position)
  playlistID INTEGER NOT NULL,
  songID INTEGER NOT NULL,
  position INTEGER NOT NULL CHECK (position >= 1),
  addedDate TEXT NOT NULL, -- 'YYYY-MM-DD'
  PRIMARY KEY (playlistID, songID),
  FOREIGN KEY (playlistID) REFERENCES PLAYLIST(playlistID)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  FOREIGN KEY (songID) REFERENCES SONG(songID)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

-- =========================
-- Snapshot table
-- =========================

CREATE TABLE LISTENING_SNAPSHOT (
  snapshotID INTEGER PRIMARY KEY,
  snapshotMonth TEXT NOT NULL, -- 'YYYY-MM-01' recommended
  totalMinutes INTEGER NOT NULL CHECK (totalMinutes >= 0),
  mostActiveDay TEXT,
  userID INTEGER NOT NULL,
  FOREIGN KEY (userID) REFERENCES USER(userID)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  UNIQUE (userID, snapshotMonth)
);

-- Helpful indexes (optional but good)
CREATE INDEX idx_album_artistID ON ALBUM(artistID);
CREATE INDEX idx_song_albumID ON SONG(albumID);
CREATE INDEX idx_playlist_userID ON PLAYLIST(userID);
CREATE INDEX idx_listen_userID ON LISTEN_HISTORY(userID);
CREATE INDEX idx_listen_songID ON LISTEN_HISTORY(songID);

PRAGMA foreign_key_list('SONG');
PRAGMA foreign_key_list('PLAYLIST_SONG');
PRAGMA foreign_key_list('LISTEN_HISTORY');
